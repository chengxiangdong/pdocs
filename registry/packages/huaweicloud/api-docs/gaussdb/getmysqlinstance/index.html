func main() {
	pulumi.Run(func(ctx *pulumi.Context) error {
		_, err := GaussDB.GetMysqlInstance(ctx, &gaussdb.GetMysqlInstanceArgs{
			Name: pulumi.StringRef("gaussdb-instance"),
		}, nil)
		if err != nil {
			return err
		}
		return nil
	})
}
