huaweicloud.Ims.getImage

Use this data source to get the ID of an available HuaweiCloud image.

Example Usage

using System.Collections.Generic;
using Pulumi;
using Huaweicloud = Pulumi.Huaweicloud;

return await Deployment.RunAsync(() => 
{
    var ubuntu = Huaweicloud.Ims.GetImage.Invoke(new()
    {
        MostRecent = true,
        Name = "Ubuntu 18.04 server 64bit",
        Visibility = "public",
    });

    var centos_1 = Huaweicloud.Ims.GetImage.Invoke(new()
    {
        Architecture = "x86",
        MostRecent = true,
        OsVersion = "CentOS 7.4 64bit",
        Visibility = "public",
    });

    var centos_2 = Huaweicloud.Ims.GetImage.Invoke(new()
    {
        Architecture = "x86",
        MostRecent = true,
        NameRegex = "^CentOS 7.4",
        Visibility = "public",
    });

    var bmsImage = Huaweicloud.Ims.GetImage.Invoke(new()
    {
        Architecture = "x86",
        ImageType = "Ironic",
        MostRecent = true,
        OsVersion = "CentOS 7.4 64bit",
        Visibility = "public",
    });

});

package main

import (
	"github.com/huaweicloud/pulumi-huaweicloud/sdk/go/huaweicloud/Ims"
	"github.com/pulumi/pulumi-huaweicloud/sdk/go/huaweicloud/Ims"
	"github.com/pulumi/pulumi/sdk/v3/go/pulumi"
)

func main() {
	pulumi.Run(func(ctx *pulumi.Context) error {
		_, err := Ims.GetImage(ctx, &ims.GetImageArgs{
			MostRecent: pulumi.BoolRef(true),
			Name:       pulumi.StringRef("Ubuntu 18.04 server 64bit"),
			Visibility: pulumi.StringRef("public"),
		}, nil)
		if err != nil {
			return err
		}
		_, err = Ims.GetImage(ctx, &ims.GetImageArgs{
			Architecture: pulumi.StringRef("x86"),
			MostRecent:   pulumi.BoolRef(true),
			OsVersion:    pulumi.StringRef("CentOS 7.4 64bit"),
			Visibility:   pulumi.StringRef("public"),
		}, nil)
		if err != nil {
			return err
		}
		_, err = Ims.GetImage(ctx, &ims.GetImageArgs{
			Architecture: pulumi.StringRef("x86"),
			MostRecent:   pulumi.BoolRef(true),
			NameRegex:    pulumi.StringRef("^CentOS 7.4"),
			Visibility:   pulumi.StringRef("public"),
		}, nil)
		if err != nil {
			return err
		}
		_, err = Ims.GetImage(ctx, &ims.GetImageArgs{
			Architecture: pulumi.StringRef("x86"),
			ImageType:    pulumi.StringRef("Ironic"),
			MostRecent:   pulumi.BoolRef(true),
			OsVersion:    pulumi.StringRef("CentOS 7.4 64bit"),
			Visibility:   pulumi.StringRef("public"),
		}, nil)
		if err != nil {
			return err
		}
		return nil
	})
}
